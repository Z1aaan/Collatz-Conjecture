 \section{Collatz Sequence for \(n=1\) }
    %Collatz Function
    \[
        f(n)=
        \begin{cases}
        \frac{n}{2}, & n \mod 2=0
        \\
        3n+1, &n \mod 2=1
        \end{cases} \\
    \]
    % Path for given N
    \(\textbf{Path for f(1)}\\[3mm]f(n), n=1
 \\ 
 \Rightarrow 3(1) + 1
 \\ 
 \Rightarrow n=4
 \\[3mm] 
f(n), n=4
 \\ 
 \Rightarrow \frac{4}{2} 
 \\ 
 \Rightarrow n=2
 \\[3mm] 
f(n), n=2
 \\ 
 \Rightarrow \frac{2}{2} 
 \\ 
 \Rightarrow n=1
 \\[3mm]
